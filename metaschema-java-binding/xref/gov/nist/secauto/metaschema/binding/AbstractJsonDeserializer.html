27  package gov.nist.secauto.metaschema.binding;

29  import com.fasterxml.jackson.core.JsonFactory;
30  import com.fasterxml.jackson.core.JsonGenerator;
31  import com.fasterxml.jackson.core.JsonParser;
32  import com.fasterxml.jackson.core.JsonToken;
33  import com.fasterxml.jackson.core.util.DefaultPrettyPrinter;
34  
35  import gov.nist.secauto.metaschema.binding.io.Configuration;
36  import gov.nist.secauto.metaschema.binding.io.Feature;
37  import gov.nist.secauto.metaschema.binding.io.json.parser.DefaultJsonParsingContext;
38  import gov.nist.secauto.metaschema.binding.io.json.parser.JsonParsingContext;
39  import gov.nist.secauto.metaschema.binding.model.AssemblyClassBinding;
40  import gov.nist.secauto.metaschema.binding.model.ClassBinding;
41  import gov.nist.secauto.metaschema.binding.model.annotations.RootWrapper;
42  
43  import org.apache.logging.log4j.LogManager;
44  import org.apache.logging.log4j.Logger;
45  
46  import java.io.IOException;
47  import java.io.Reader;
48  import java.io.Writer;
49  
50  abstract class AbstractJsonDeserializer<CLASS> extends AbstractDeserializer<CLASS> {
51    private static final Logger logger = LogManager.getLogger(AbstractJsonDeserializer.class);
52  
53    private JsonFactory jsonFactory;
54  
55    public AbstractJsonDeserializer(BindingContext bindingContext, AssemblyClassBinding<CLASS> classBinding,
56        Configuration configuration) {
57      super(bindingContext, classBinding, configuration);
58    }
59  
60    protected abstract JsonFactory getJsonFactoryInstance();
61  
62    protected JsonFactory getJsonFactory() {
63      synchronized (this) {
64        if (jsonFactory == null) {
65          jsonFactory = getJsonFactoryInstance();
66        }
67        return jsonFactory;
68      }
69    }
70  
71    protected void setJsonFactory(JsonFactory jsonFactory) {
72      synchronized (this) {
73        this.jsonFactory = jsonFactory;
74      }
75    }
76  
77    protected JsonParser newJsonParser(Reader reader) throws BindingException {
78      try {
79        JsonFactory factory = getJsonFactory();
80        JsonParser retval = factory.createParser(reader);
81        return retval;
82      } catch (IOException ex) {
83        throw new BindingException(ex);
84      }
85    }
86  
87    protected JsonGenerator newJsonGenerator(Writer writer) throws BindingException {
88      try {
89        JsonFactory factory = getJsonFactory();
90        JsonGenerator retval = factory.createGenerator(writer);
91        retval.setPrettyPrinter(new DefaultPrettyPrinter());
92        return retval;
93      } catch (IOException ex) {
94        throw new BindingException(ex);
95      }
96    }
97  
98    @Override
99    public CLASS deserialize(Reader reader) throws BindingException {
100     JsonParser parser = newJsonParser(reader);
101 
102     Class<CLASS> clazz = getClassBinding().getClazz();
103 
104     JsonToken token;
105     try {
106       token = parser.nextToken();
107     } catch (IOException ex) {
108       throw new BindingException("Unable to read JSON.", ex);
109     }
110     if (!parser.isExpectedStartObjectToken()) {
111       throw new BindingException(String.format("Start object expected. Found '%s'.", token.toString()));
112     }
113 
114     ClassBinding<CLASS> classBinding = getClassBinding();
115     boolean isParseRoot = false;
116     if (getConfiguration().isFeatureEnabled(Feature.DESERIALIZE_ROOT, false)) {
117       if (classBinding.hasRootWrapper()) {
118         isParseRoot = true;
119       } else if (logger.isDebugEnabled()) {
120         logger.debug(
121             String.format("Root parsing is enabled (DESERIALIZE_ROOT), but class '%s' is missing a '%s' annotation.",
122                 clazz.getName(), RootWrapper.class.getName()));
123       }
124     }
125 
126     BindingContext bindingContext = getBindingContext();
127     JsonParsingContext parsingContext = new DefaultJsonParsingContext(parser, bindingContext);
128     return classBinding.getJsonReader(bindingContext).readJson(parsingContext, null, isParseRoot).get(0);
129   }
