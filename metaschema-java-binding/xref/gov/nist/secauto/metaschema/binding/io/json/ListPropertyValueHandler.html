27  package gov.nist.secauto.metaschema.binding.io.json;

29  import com.fasterxml.jackson.core.JsonParser;
30  import com.fasterxml.jackson.core.JsonToken;
31  
32  import gov.nist.secauto.metaschema.binding.BindingException;
33  import gov.nist.secauto.metaschema.binding.io.json.parser.JsonParsingContext;
34  import gov.nist.secauto.metaschema.binding.io.json.parser.JsonUtil;
35  import gov.nist.secauto.metaschema.binding.model.ClassBinding;
36  
37  import java.io.IOException;
38  import java.util.LinkedList;
39  import java.util.List;
40  import java.util.function.Supplier;
41  
42  public class ListPropertyValueHandler extends AbstractPropertyValueHandler {
43    private final boolean allowSingletonValue;
44    private final List<Object> values = new LinkedList<>();
45    private boolean firstValueParsed = false;
46  
47    public ListPropertyValueHandler(ClassBinding<?> classBinding, PropertyItemHandler propertyItemHandler,
48        boolean requireSingletonValue) {
49      super(classBinding, propertyItemHandler);
50      this.allowSingletonValue = requireSingletonValue;
51    }
52  
53    protected boolean isRequireSingletonValue() {
54      return allowSingletonValue;
55    }
56  
57    @Override
58    public boolean parseNextFieldValue(JsonParsingContext parsingContext) throws BindingException, IOException {
59      JsonParser parser = parsingContext.getEventReader();
60      JsonToken currentToken = parser.currentToken();
61  
62      if (!firstValueParsed) {
63        if (JsonToken.START_ARRAY.equals(currentToken)) {
64          // advance to the start of the first value item
65          currentToken = parser.nextToken();
66        } else if (!allowSingletonValue) {
67          throw new BindingException(
68              String.format("Found unexpected '%s' token when parsing %s. This list doesn't allow a singleton object.",
69                  currentToken, JsonUtil.toLocationContext(parser, getClassBinding(), getPropertyBinding())));
70        }
71        firstValueParsed = true;
72      }
73  
74      // Parse the value at the current token; after parsing the current token is
75      // expected to be at the next START_OBJECT, END_ARRAY, or FIELD_NAME
76      PropertyItemHandler propertyItemHandler = getPropertyItemHandler();
77      List<Object> parsedValues = propertyItemHandler.parse(parsingContext, null);
78  
79      // if (!JsonUtil.checkEndOfValue(parser, currentToken)) {
80      // throw new BindingException(String.format("Unexpected end state token '%s' after parsing %s.",
81      // currentToken,
82      // JsonUtil.toLocationContext(parser, getClassBinding(), getPropertyBinding())));
83      // }
84      //
85      // // advance past the end
86      currentToken = parser.currentToken();
87  
88      boolean retval;
89      if (parsedValues != null) {
90        values.addAll(parsedValues);
91      }
92  
93      if (JsonToken.END_ARRAY.equals(currentToken)) {
94        if (!firstValueParsed) {
95          throw new BindingException(String.format("Found unexpected END_ARRAY token when parsing %s.",
96              JsonUtil.toLocationContext(parser, getClassBinding(), getPropertyBinding())));
97        } else {
98          retval = false;
99          // skip past the END_ARRAY
100         currentToken = parser.nextToken();
101       }
102     } else if (JsonToken.END_OBJECT.equals(currentToken) || JsonToken.FIELD_NAME.equals(currentToken)) {
103       retval = false;
104     } else {
105       retval = true;
106     }
107     return retval;
108   }
109 
110   @Override
111   public Supplier<List<Object>> getObjectSupplier() {
112     return () -> values;
113   }
114 }
