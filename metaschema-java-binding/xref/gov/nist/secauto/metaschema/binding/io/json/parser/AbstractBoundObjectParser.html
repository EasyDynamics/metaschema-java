53    public AbstractBoundObjectParser(CLASS_BINDING classBinding, JsonParsingContext parsingContext,
54        UnknownPropertyHandler unknownPropertyHandler) {

57      Objects.requireNonNull(unknownPropertyHandler, "unknownPropertyHandler");

60      this.parsingContext = parsingContext;
61      this.unknownPropertyHandler = unknownPropertyHandler;
62    }

85    protected void parseProperties(Map<String, PropertyBinding> propertyBindings) throws IOException, BindingException {
86      JsonParsingContext parsingContext = getParsingContext();
87      JsonParser parser = parsingContext.getEventReader();
88      JsonProblemHandler problemHandler = parsingContext.getProblemHandler();
89      CLASS_BINDING classBinding = getClassBinding();
90  
91      PropertyBindingSupplier propertyBindingSupplier = getPropertyBindingSupplier();
92      PropertyAccessorSupplier propertyAccessorSupplier = getPropertyAccessorSupplier();
93      PropertyParser propertyParser = getPropertyParser();
94  
95      // we are at a start object
96      JsonUtil.expectCurrentToken(parser, JsonToken.START_OBJECT);
97      // parse to the first field
98      parser.nextToken();
99  
100     Set<String> parsedProperties = new HashSet<>();
101     Set<String> unknownProperties = new HashSet<>();
102     String nextFieldName;
103     while (JsonToken.FIELD_NAME.equals(parser.currentToken()) && (nextFieldName = parser.getCurrentName()) != null) {
104 
105       // JsonUtil.expectCurrentToken(parser, JsonToken.FIELD_NAME);
106       // parse to "value" token, which may be a START_ARRAY, START_OBJECT, or one of
107       // the "VALUE" tokens
108       parser.nextToken();
109 
110       PropertyBinding propertyBinding = propertyBindings.get(nextFieldName);
111       if (propertyBinding == null) {
112         // the field is not bound
113 
114         // Give the problem handler a chance to handle the property
115         if (problemHandler.canHandleUnknownProperty(classBinding, nextFieldName, parsingContext)) {
116           for (Map.Entry<PropertyAccessor, Supplier<?>> entry : problemHandler
117               .handleUnknownProperty(classBinding, nextFieldName, parsingContext).entrySet()) {
118             propertyAccessorSupplier.apply(entry.getKey(), entry.getValue());
119           }
120         } else {
121           // handle it internally if possible (e.g., valueKey with flag)
122           Map<PropertyBinding, Supplier<?>> result = getUnknownPropertyHandler().handle(nextFieldName,
123               Collections.unmodifiableSet(unknownProperties), parsingContext);
124           for (Map.Entry<PropertyBinding, Supplier<?>> entry : result.entrySet()) {
125             propertyBindingSupplier.apply(entry.getKey(), entry.getValue());
126           }
127 
128           unknownProperties.add(nextFieldName);
129         }
130       } else {
131         // logger.info(" Parsing field '{}'", nextFieldName);
132 
133         // if the callback doesn't opt to parse the property, parse it here
134         if (!propertyParser.parse(propertyBinding)) {
135           Supplier<? extends Object> supplier = parseProperty(propertyBinding, parsingContext);
136           propertyBindingSupplier.apply(propertyBinding, supplier);
137         }
138         parsedProperties.add(nextFieldName);
139       }
140     }
141 
142     // Parse to after the end object
143     JsonUtil.expectCurrentToken(parser, JsonToken.END_OBJECT);
144     parser.nextToken();
145 
146     Map<String, PropertyBinding> missingPropertyBindings = new HashMap<>(propertyBindings);
147     Set<String> keySet = missingPropertyBindings.keySet();
148     keySet.removeAll(parsedProperties);
149     for (Map.Entry<PropertyBinding, Supplier<?>> entry : problemHandler
150         .handleMissingFields(classBinding, missingPropertyBindings, parsingContext).entrySet()) {
151       propertyBindingSupplier.apply(entry.getKey(), entry.getValue());
152     }
153 
154     // for (String key : keySet) {
155     // logger.info(" Unparsed field '{}'", key);
156     // }
157   }

159   /**
160    * Parses the field and value for the bound property provided by the {@code propertyBinding}
161    * parameter.
162    * <p>
163    * When called, the parser's current token is expected to be at the VALUE.
164    * <p>
165    * When this method returns, the parser's current token is expected to be at the next
166    * {@link JsonToken#FIELD_NAME} if there are more fields to parse, or at the
167    * {@link JsonToken#END_OBJECT} if the last field has been parsed.
168    * 
169    * @param propertyBinding
170    *          the bound property to be parsed
171    * @param parsingContext
172    *          the configured parsing context
173    * @return a supplier that will provide the value
174    * @throws BindingException
175    *           if a binding error occurs whole parsing the target property
176    * @throws IOException
177    *           if an input error has occurred
178    */
179   public Supplier<? extends Object> parseProperty(PropertyBinding propertyBinding, JsonParsingContext parsingContext)
180       throws BindingException, IOException {
181 
182     PropertyValueHandler propertyValueHandler
183         = PropertyValueHandler.newPropertyValueHandler(getClassBinding(), propertyBinding, parsingContext);
184 
185     while (propertyValueHandler.parseNextFieldValue(parsingContext)) {
186       // after calling parseNextField the current token is expected to be at the next
187       // field to parse or at the END_OBJECT for the containing object
188     }
189 
190     return propertyValueHandler.getObjectSupplier();
191   }
