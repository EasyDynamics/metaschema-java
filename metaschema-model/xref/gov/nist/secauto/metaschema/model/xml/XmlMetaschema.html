69    /**
70     * Loads a metaschema from the provided resource.
71     * <p>
72     * If the metaschema imports other metaschema, the provided visitedMetaschema can be used to track
73     * circular inclusions. This is useful when this method recurses into included metaschema.
74     * <p>
75     * Previously loaded metaschema are provided by the metaschemaCache. This method will add the
76     * current metaschema to the cache after all imported metaschema have been loaded.
77     * 
78     * @param resource
79     *          the metaschema resource to load
80     * @param visitedMetaschema
81     *          a LIFO queue representing previously visited metaschema in an import chain
82     * @param metaschemaCache
83     *          a map of previously processed metaschema, keyed by the resource URI of the metaschema
84     * @return the loaded metaschema
85     * @throws MetaschemaException
86     *           if a parsing or processing error occurs
87     * @throws MalformedURLException
88     *           if the provided URI is malformed
89     * @throws IOException
90     *           if an IO error occurs while parsing
91     */
92    public static XmlMetaschema loadMetaschema(URI resource, Stack<URI> visitedMetaschema,
93        Map<URI, XmlMetaschema> metaschemaCache) throws MetaschemaException, MalformedURLException, IOException {
94      // first check if the current Metaschema has been visited to prevent cycles
95      if (visitedMetaschema.contains(resource)) {
96        throw new MetaschemaException("Cycle detected in metaschema includes for '" + resource + "'. Call stack: '"
97            + visitedMetaschema.stream().map(n -> n.toString()).collect(Collectors.joining(",")));
98      }
99  
100     XmlMetaschema retval = metaschemaCache.get(resource);
101     if (retval == null) {
102       logger.info("Loading metaschema '{}'", resource);
103       // parse this metaschema
104       METASCHEMADocument metaschemaXml;
105       try {
106         metaschemaXml = METASCHEMADocument.Factory.parse(resource.toURL());
107       } catch (XmlException e) {
108         throw new MetaschemaException(e);
109       }
110 
111       // now check if this Metaschema imports other metaschema
112       int size = metaschemaXml.getMETASCHEMA().sizeOfImportArray();
113       Map<URI, Metaschema> importedMetaschema;
114       if (size == 0) {
115         importedMetaschema = Collections.emptyMap();
116       } else {
117         visitedMetaschema.push(resource);
118         try {
119           importedMetaschema = new LinkedHashMap<>();
120           for (ImportDocument.Import imported : metaschemaXml.getMETASCHEMA().getImportList()) {
121             URI importedResource = URI.create(imported.getHref());
122             importedResource = resource.resolve(importedResource);
123             importedMetaschema.put(importedResource,
124                 loadMetaschema(importedResource, visitedMetaschema, metaschemaCache));
125           }
126         } finally {
127           visitedMetaschema.pop();
128         }
129       }
130 
131       // now create this metaschema
132       retval = new XmlMetaschema(resource, metaschemaXml, importedMetaschema);
133       metaschemaCache.put(resource, retval);
134     } else {
135       logger.debug("Found metaschema in cache '{}'", resource);
136     }
137     return retval;
138   }

140   protected static ModelBindingConfiguration getBindingConfiguration(METASCHEMADocument metaschema) {
141     ModelBindingConfiguration retval = null;
142     METASCHEMA modelMetaschema = metaschema.getMETASCHEMA();
143     if (modelMetaschema.isSetExtensions()) {
144       Extensions extensions = modelMetaschema.getExtensions();
145       for (ExtensionType extensionInstance : extensions.getModelExtensionList()) {
146         System.out.println("Extension Class: " + extensionInstance.getClass().getName());
147         if (extensionInstance instanceof ModelBindingDocument.ModelBinding) {
148           ModelBindingDocument.ModelBinding modelConfig = (ModelBindingDocument.ModelBinding) extensionInstance;
149           if (modelConfig.isSetJava()) {
150             ModelBindingDocument.ModelBinding.Java modelJava = modelConfig.getJava();
151             retval = new ModelBindingConfiguration(modelJava.getPackageName());
152             break;
153           }
154         }
155       }
156     }
157 
158     if (retval == null) {
159       retval = ModelBindingConfiguration.NULL_CONFIG;
160     }
161     return retval;
162   }

170   /**
171    * Constructs a new Metaschema instance.
172    * 
173    * @param resource
174    *          the resource from which the metaschema was loaded
175    * @param metaschemaXml
176    *          the XML source of the metaschema definition bound to Java objects
177    * @param importedMetaschema
178    *          the definitions for any metaschema imported by this metaschema
179    * @throws MetaschemaException
180    *           if a processing error occurs
181    */
182   public XmlMetaschema(URI resource, METASCHEMADocument metaschemaXml, Map<URI, Metaschema> importedMetaschema)
183       throws MetaschemaException {
184     super(resource, getBindingConfiguration(metaschemaXml), importedMetaschema);
185     this.metaschema = metaschemaXml;
186 
187     XmlCursor cursor = metaschema.getMETASCHEMA().newCursor();
188     cursor.selectPath("declare namespace m='http://csrc.nist.gov/ns/oscal/metaschema/1.0';"
189         + "$this/m:define-assembly|$this/m:define-field|$this/m:define-flag");
190 
191     // handle definitions in this metaschema
192     Map<String, InfoElementDefinition> infoElementDefinitions = new LinkedHashMap<>();
193     Map<String, XmlFlagDefinition> flagDefinitions = new LinkedHashMap<>();
194     Map<String, XmlFieldDefinition> fieldDefinitions = new LinkedHashMap<>();
195     Map<String, XmlAssemblyDefinition> assemblyDefinitions = new LinkedHashMap<>();
196 
197     while (cursor.toNextSelection()) {
198       XmlObject obj = cursor.getObject();
199       if (obj instanceof DefineFlagDocument.DefineFlag) {
200         XmlFlagDefinitionel/xml/XmlFlagDefinition.html#XmlFlagDefinition">XmlFlagDefinition flag = new XmlFlagDefinition((DefineFlagDocument.DefineFlag) obj, this);
201         logger.trace("New flag definition '{}'", Util.toCoordinates(flag));
202         flagDefinitions.put(flag.getName(), flag);
203         infoElementDefinitions.put(flag.getName(), flag);
204       } else if (obj instanceof DefineFieldDocument.DefineField) {
205         XmlFieldDefinition/xml/XmlFieldDefinition.html#XmlFieldDefinition">XmlFieldDefinition field = new XmlFieldDefinition((DefineFieldDocument.DefineField) obj, this);
206         logger.trace("New field definition '{}'", Util.toCoordinates(field));
207         fieldDefinitions.put(field.getName(), field);
208         infoElementDefinitions.put(field.getName(), field);
209       } else if (obj instanceof DefineAssemblyDocument.DefineAssembly) {
210         XmlAssemblyDefinitionmlAssemblyDefinition.html#XmlAssemblyDefinition">XmlAssemblyDefinition assembly = new XmlAssemblyDefinition((DefineAssemblyDocument.DefineAssembly) obj, this);
211         logger.trace("New assembly definition '{}'", Util.toCoordinates(assembly));
212         assemblyDefinitions.put(assembly.getName(), assembly);
213         infoElementDefinitions.put(assembly.getName(), assembly);
214       }
215     }
216     this.flagDefinitions
217         = flagDefinitions.isEmpty() ? Collections.emptyMap() : Collections.unmodifiableMap(flagDefinitions);
218     this.fieldDefinitions
219         = fieldDefinitions.isEmpty() ? Collections.emptyMap() : Collections.unmodifiableMap(fieldDefinitions);
220     this.assemblyDefinitions
221         = assemblyDefinitions.isEmpty() ? Collections.emptyMap() : Collections.unmodifiableMap(assemblyDefinitions);
222     this.infoElementDefinitions = infoElementDefinitions.isEmpty() ? Collections.emptyMap()
223         : Collections.unmodifiableMap(infoElementDefinitions);
224 
225     parseUsedDefinitions();
226   }
