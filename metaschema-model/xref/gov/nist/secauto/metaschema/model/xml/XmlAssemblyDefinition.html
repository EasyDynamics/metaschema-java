58    protected static AssemblyBindingConfiguration
59        getBindingConfiguration(DefineAssemblyDocument.DefineAssembly xmlField) {
60      AssemblyBindingConfiguration retval = null;
61      if (xmlField.isSetExtensions()) {
62        DefineAssemblyDocument.DefineAssembly.Extensions extensions = xmlField.getExtensions();
63        for (ExtensionType extensionInstance : extensions.getDefineAssemblyExtensionList()) {
64          System.out.println("Extension Class: " + extensionInstance.getClass().getName());
65          if (extensionInstance instanceof DefineAssemblyBindingDocument.DefineAssemblyBinding) {
66            DefineAssemblyBindingDocument.DefineAssemblyBinding modelConfig
67                = (DefineAssemblyBindingDocument.DefineAssemblyBinding) extensionInstance;
68            if (modelConfig.isSetJava()) {
69              DefineAssemblyBindingDocument.DefineAssemblyBinding.Java modelJava = modelConfig.getJava();
70  
71              retval = new AssemblyBindingConfiguration(modelJava.getClassName(), modelJava.getBaseClassName(),
72                  modelJava.getInterfaceNameList());
73              break;
74            }
75          }
76        }
77      }
78  
79      if (retval == null) {
80        retval = AssemblyBindingConfiguration.NULL_CONFIG;
81      }
82      return retval;
83    }

92    /**
93     * Constructs a new Metaschema Assembly definition from an XML representation bound to Java objects.
94     * 
95     * @param xmlAssembly
96     *          the XML representation bound to Java objects
97     * @param metaschema
98     *          the containing Metaschema
99     */
100   public XmlAssemblyDefinition(DefineAssemblyDocument.DefineAssembly xmlAssembly, XmlMetaschema metaschema) {
101     super(getBindingConfiguration(xmlAssembly), metaschema);
102     this.xmlAssembly = xmlAssembly;
103     //
104     // MarkupStringConverter.toMarkupString(getXmlAssembly().getDescription());
105 
106     Map<String, InfoElementInstance> infoElementInstances = new LinkedHashMap<>();
107     int numFlags = xmlAssembly.sizeOfFlagArray();
108     if (numFlags > 0) {
109       Map<String, XmlFlagInstance> flagInstances = new LinkedHashMap<>();
110       for (FlagDocument.Flag xmlFlag : xmlAssembly.getFlagList()) {
111         XmlFlagInstance/XmlFlagInstance.html#XmlFlagInstance">XmlFlagInstance flagInstance = new XmlFlagInstance(xmlFlag, this);
112         flagInstances.put(flagInstance.getName(), flagInstance);
113         infoElementInstances.put(flagInstance.getName(), flagInstance);
114       }
115       this.flagInstances = Collections.unmodifiableMap(flagInstances);
116     } else {
117       flagInstances = Collections.emptyMap();
118     }
119 
120     XmlCursor cursor = xmlAssembly.getModel().newCursor();
121     cursor.selectPath("declare namespace m='http://csrc.nist.gov/ns/oscal/metaschema/1.0';"
122         + "$this/m:assembly|$this/m:field|$this/m:choice");
123 
124     Map<String, XmlFieldInstance> fieldInstances = new LinkedHashMap<>();
125     Map<String, XmlAssemblyInstance> assemblyInstances = new LinkedHashMap<>();
126     List<InfoElementInstance> modelInstances = new ArrayList<>(cursor.getSelectionCount());
127     Map<String, ModelInstance> namedModelInstances = new LinkedHashMap<>();
128     while (cursor.toNextSelection()) {
129       XmlObject obj = cursor.getObject();
130       if (obj instanceof FieldDocument.Field) {
131         XmlFieldInstanceel/xml/XmlFieldInstance.html#XmlFieldInstance">XmlFieldInstance field = new XmlFieldInstance((FieldDocument.Field) obj, this);
132         fieldInstances.put(field.getName(), field);
133         infoElementInstances.put(field.getName(), field);
134         modelInstances.add(field);
135         namedModelInstances.put(field.getName(), field);
136       } else if (obj instanceof AssemblyDocument.Assembly) {
137         XmlAssemblyInstance/XmlAssemblyInstance.html#XmlAssemblyInstance">XmlAssemblyInstance assembly = new XmlAssemblyInstance((AssemblyDocument.Assembly) obj, this);
138         assemblyInstances.put(assembly.getName(), assembly);
139         infoElementInstances.put(assembly.getName(), assembly);
140         modelInstances.add(assembly);
141         namedModelInstances.put(assembly.getName(), assembly);
142       } else if (obj instanceof ChoiceDocument.Choice) {
143         XmlChoiceInstance/xml/XmlChoiceInstance.html#XmlChoiceInstance">XmlChoiceInstance choice = new XmlChoiceInstance((ChoiceDocument.Choice) obj, this);
144         assemblyInstances.putAll(choice.getAssemblyInstances());
145         fieldInstances.putAll(choice.getFieldInstances());
146         infoElementInstances.putAll(choice.getNamedModelInstances());
147         modelInstances.add(choice);
148       }
149 
150     }
151 
152     this.namedModelInstances
153         = infoElementInstances.isEmpty() ? Collections.emptyMap() : Collections.unmodifiableMap(namedModelInstances);
154     this.fieldInstances
155         = fieldInstances.isEmpty() ? Collections.emptyMap() : Collections.unmodifiableMap(fieldInstances);
156     this.assemblyInstances
157         = assemblyInstances.isEmpty() ? Collections.emptyMap() : Collections.unmodifiableMap(assemblyInstances);
158     this.modelInstances
159         = modelInstances.isEmpty() ? Collections.emptyList() : Collections.unmodifiableList(modelInstances);
160   }

199     // this shouldn't get called all that often, so this is better than allocating memory
200     return modelInstances.stream().filter(p -> Type.CHOICE.equals(p.getType())).map(obj -> (ChoiceInstance) obj)
