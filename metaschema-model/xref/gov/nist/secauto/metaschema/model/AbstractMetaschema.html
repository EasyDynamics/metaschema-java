76        Map<URI, ? extends Metaschema> importedMetaschema) {
77      Objects.requireNonNull(metaschemaResource, "metaschemaResource");

83      logger.trace("Creating metaschema '{}'", metaschemaResource);

115   protected void parseUsedDefinitions() throws MetaschemaException {
116     logger.debug("Processing metaschema '{}'", this.getLocation());
117 
118     // handle definitions used in this metaschema, including any referenced definitions in included
119     // metaschema
120     if (getImportedMetaschema().isEmpty()) {
121       this.usedInfoElementDefinitions = new LinkedHashMap<>(getInfoElementDefinitions());
122       this.usedFlagDefinitions = new LinkedHashMap<>(getFlagDefinitions());
123       this.usedFieldDefinitions = new LinkedHashMap<>(getFieldDefinitions());
124       this.usedAssemblyDefinitions = new LinkedHashMap<>(getAssemblyDefinitions());
125     } else {
126       AssemblyDefinition rootAssemblyDefinition = getRootAssemblyDefinition();
127       Map<String, InfoElementDefinition> usedDefinitions = new LinkedHashMap<>();
128       processUsedDefinitions(rootAssemblyDefinition, usedDefinitions);
129 
130       this.usedInfoElementDefinitions = new LinkedHashMap<>();
131       this.usedFlagDefinitions = new LinkedHashMap<>();
132       this.usedFieldDefinitions = new LinkedHashMap<>();
133       this.usedAssemblyDefinitions = new LinkedHashMap<>();
134 
135       List<Metaschema> metaschemas = new ArrayList<>(getImportedMetaschema().size() + 1);
136       metaschemas.addAll(getImportedMetaschema().values());
137       metaschemas.add(this);
138 
139       for (Metaschema metaschema : metaschemas) {
140         for (InfoElementDefinition definition : metaschema.getInfoElementDefinitions().values()) {
141           logger.trace("Checking use of '{}'", Util.toCoordinates(definition));
142 
143           InfoElementDefinition usedDefinition = usedDefinitions.get(definition.getName());
144           boolean used = definition.equals(usedDefinition);
145           if (used) {
146             logger.debug("Using definition '{}'", Util.toCoordinates(usedDefinition));
147           } else {
148             if (usedDefinition != null) {
149               logger.trace("  Unused definition '{}'", Util.toCoordinates(usedDefinition));
150             } else {
151               logger.trace("  No match for definition '{}'", Util.toCoordinates(definition));
152             }
153             continue;
154           }
155 
156           String usedName = usedDefinition.getName();
157           this.usedInfoElementDefinitions.put(usedName, usedDefinition);
158           if (usedDefinition instanceof FlagDefinition) {
159             this.usedFlagDefinitions.put(usedName, (FlagDefinition) usedDefinition);
160           } else if (usedDefinition instanceof FieldDefinition) {
161             this.usedFieldDefinitions.put(usedName, (FieldDefinition) usedDefinition);
162           } else if (usedDefinition instanceof AssemblyDefinition) {
163             this.usedAssemblyDefinitions.put(usedName, (AssemblyDefinition) usedDefinition);
164           }
165         }
166       }
167     }
168 
169   }

171   private void processUsedDefinitions(AssemblyDefinition definition, Map<String, InfoElementDefinition> usedDefinitions)
172       throws MetaschemaException {
173     logger.trace("Analyzing use of assembly '{}'", Util.toCoordinates(definition));
174     if (isNewDefinition(definition, usedDefinitions)) {
175       usedDefinitions.put(definition.getName(), definition);
176 
177       // get all referenced flags
178       processUsedDefinitions((FlagContainer) definition, usedDefinitions);
179 
180       // get all referenced assemblies and fields, and anything they reference
181       processUsedDefinitions((ModelContainer) definition, usedDefinitions);
182     } else {
183       logger.trace("  skipping assembly '{}'", Util.toCoordinates(definition));
184     }
185   }

187   private void processUsedDefinitions(FlagContainer container, Map<String, InfoElementDefinition> usedDefinitions)
188       throws MetaschemaException {
189     // get all referenced flags
190     for (FlagInstance instance : container.getFlagInstances().values()) {
191       logger.trace("Analyzing use of flag '{}'", Util.toCoordinates(instance));
192       if (instance.isReference()) {
193         FlagDefinition flagDefinition = instance.getDefinition();
194         if (isNewDefinition(flagDefinition, usedDefinitions)) {
195           usedDefinitions.put(flagDefinition.getName(), flagDefinition);
196           logger.debug("Using definition '{}'", Util.toCoordinates(flagDefinition));
197         } else {
198           logger.debug("  Skipping known definition '{}'", Util.toCoordinates(flagDefinition));
199         }
200       } else {
201         // a local definition
202         logger.debug("  Skipping local instance '{}'", Util.toCoordinates(instance));
203       }
204     }
205   }

207   private void processUsedDefinitions(ModelContainer modelContainer, Map<String, InfoElementDefinition> usedDefinitions)
208       throws MetaschemaException {
209     for (InfoElementInstance instance : modelContainer.getInstances()) {
210       logger.trace("Processing model instance '{}'", Util.toCoordinates(instance));
211       InfoElementDefinition definition = instance.getDefinition();
212       if (instance.isReference()) {
213         if (!isNewDefinition(definition, usedDefinitions)) {
214           logger.debug("  Skipping known definition '{}'", Util.toCoordinates(definition));
215           // skip this one, since we have already processed it
216           continue;
217         } else {
218           // handle below
219         }
220       } else {
221         // a local definition, we still need to process its instances
222       }
223 
224       switch (instance.getType()) {
225       case FLAG:
226         // no child instances
227         break;
228       case FIELD:
229         usedDefinitions.put(definition.getName(), definition);
230         logger.debug("Using definition '{}'", Util.toCoordinates(definition));
231         FieldInstance/gov/nist/secauto/metaschema/model/info/instances/FieldInstance.html#FieldInstance">FieldInstance fieldInstance = (FieldInstance) instance;
232         processUsedDefinitions((FlagContainer) fieldInstance.getDefinition(), usedDefinitions);
233         break;
234       case ASSEMBLY:
235         AssemblyInstanceist/secauto/metaschema/model/info/instances/AssemblyInstance.html#AssemblyInstance">AssemblyInstance assemblyInstance = (AssemblyInstance) instance;
236         processUsedDefinitions(assemblyInstance.getDefinition(), usedDefinitions);
237         break;
238       case CHOICE:
239         ChoiceInstanceov/nist/secauto/metaschema/model/info/instances/ChoiceInstance.html#ChoiceInstance">ChoiceInstance choiceInstance = (ChoiceInstance) instance;
240         processUsedDefinitions((ModelContainer) choiceInstance, usedDefinitions);
241         break;
242       default:
243         throw new RuntimeException("Unexpected type: " + instance.getType());
244       }
245     }
246   }

248   private boolean isNewDefinition(InfoElementDefinition definition, Map<String, InfoElementDefinition> usedDefinitions)
249       throws MetaschemaException {
250     boolean processElement = true;
251     if (usedDefinitions.containsKey(definition.getName())) {
252       InfoElementDefinition existingDefinition = usedDefinitions.get(definition.getName());
253       if (!existingDefinition.getType().equals(definition.getType())) {
254         // check if the types differ
255         throw (new MetaschemaException(
256             String.format("New element '%s: (%s) %s' has conflicting type with old element '%s:(%s)%s'",
257                 definition.getContainingMetaschema().getLocation(), definition.getType(), definition.getName(),
258                 existingDefinition.getContainingMetaschema().getLocation(), existingDefinition.getType(),
259                 existingDefinition.getName())));
260       } else if (!existingDefinition.equals(definition)) {
261         // the field or assembly is shadowing an existing definition
262         throw (new MetaschemaException(String.format("New element '%s: (%s) %s' shadows old element '%s:(%s)%s'",
263             definition.getContainingMetaschema().getLocation(), definition.getType(), definition.getName(),
264             existingDefinition.getContainingMetaschema().getLocation(), existingDefinition.getType(),
265             existingDefinition.getName())));
266       }
267       processElement = false;
268     }
269     return processElement;
270   }

277   @Override
278   public InfoElementDefinition getInfoElementDefinitionByName(String name) {
279     InfoElementDefinition retval = getInfoElementDefinitions().get(name);
280     if (retval == null) {
281       for (Metaschema metaschema : getImportedMetaschema().values()) {
282         if ((retval = metaschema.getInfoElementDefinitionByName(name)) != null) {
283           break;
284         }
285       }
286     }
287     return retval;
288   }
