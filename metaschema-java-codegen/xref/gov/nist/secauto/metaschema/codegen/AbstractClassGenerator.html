60    /**
61     * Constructs a new class generator based on the provided definition.
62     * 
63     * @param definition
64     *          the Metaschema field or assembly definition for which a Java class will be generated
65     * @param supplier
66     *          a callback used to generate class type information for a generated class
67     */
68    public AbstractClassGenerator(DEFINITION definition, JavaTypeSupplier supplier) {
69      Objects.requireNonNull(definition, "definition");
70      Objects.requireNonNull(supplier, "supplier");
71      this.definition = definition;
72      this.supplier = supplier;
73  
74      this.javaType = supplier.getClassJavaType(getDefinition());
75      for (FlagInstance instance : definition.getFlagInstances().values()) {
76        newFlagPropertyGenerator(instance);
77      }
78    }

143   /**
144    * Adds the provided property generator to this class generator.
145    * 
146    * @param property
147    *          the property generator to add
148    */
149   protected void addPropertyGenerator(PropertyGenerator property) {
150     String name = property.getPropertyName();
151     PropertyGenerator oldContext = propertyNameToPropertyGeneratorMap.put(name, property);
152     if (oldContext != null) {
153       logger.error("Unexpected duplicate instance property name '{}'", name);
154       throw new RuntimeException(String.format("Unexpected duplicate instance property name '%s'", name));
155     }
156   }
