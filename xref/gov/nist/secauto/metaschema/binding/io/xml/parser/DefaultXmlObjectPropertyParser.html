48  public class DefaultXmlObjectPropertyParser extends AbstractXmlPropertyParser<ModelItemPropertyBinding>
49      implements XmlObjectPropertyParser {
50    private static final Logger logger = LogManager.getLogger(DefaultXmlObjectPropertyParser.class);
51  
52    private final QName groupWrapperQName;
53    private final QName itemWrapperQName;
54    private JavaTypeAdapter<?> typeAdapter;
55  
56    public DefaultXmlObjectPropertyParser(ModelItemPropertyBinding propertyBinding, BindingContext bindingContext) {
57      super(propertyBinding, bindingContext);
58      groupWrapperQName = getGroupWrapperQName();
59      itemWrapperQName = getItemWrapperQName();
60    }
61  
62    @Override
63    public <CLASS> void parse(CLASS obj, XmlParsingContext parsingContext) throws BindingException {
64      PropertyCollector collector = getPropertyBinding().getPropertyInfo().newPropertyCollector();
65      XMLEventReader2 reader = parsingContext.getEventReader();
66  
67      if (groupWrapperQName != null) {
68        consumeStartElement(reader, groupWrapperQName);
69      }
70  
71      JavaTypeAdapter<?> typeAdapter = getTypeAdapter();
72      if (logger.isDebugEnabled()) {
73        logger.debug("Using Adapter '{}'", typeAdapter.getClass().getName());
74      }
75  
76      XMLEvent nextEvent;
77      try {
78        // continue while there is more work to do
79        do {
80          // skip inter-element whitespace
81          nextEvent = XmlEventUtil.skipWhitespace(reader);
82  
83          /*
84           * In most cases, items will be wrapped. The only exception is an Assembly with a MarkupMultiline
85           * field with inXmlWrapped equal to {@code false}. In the latter case that object will be parsed in
86           * a single loop.
87           */
88          if (itemWrapperQName != null && !typeAdapter.isParsingStartElement()) {
89            consumeStartElement(reader, itemWrapperQName);
90          }
91  
92          // Pass in the start element if it exists to allow attributes to be parsed
93          Object value = typeAdapter.parse(parsingContext);
94          if (value != null) {
95            collector.add(value);
96          } else {
97            // TODO: should we do something here? Maybe a problem handler?
98          }
99  
100         nextEvent = reader.peek();
101         if (itemWrapperQName != null && !typeAdapter.isParsingStartElement()) {
102           assert XmlEventUtil.isNextEventEndElement(reader, itemWrapperQName) : XmlEventUtil.toString(nextEvent);
103           consumeEndElement(reader, itemWrapperQName);
104         } else {
105           // otherwise, the type adapter is expected to advance to the next child's
106           // START_ELEMENT or the parent's END_ELEMENT. We need to trust that the type
107           // adapter has not parsed too far.
108         }
109         nextEvent = XmlEventUtil.skipWhitespace(reader);
110 
111         // do not continue if the object is MultilineMarkup
112       } while (itemWrapperQName != null && XmlEventUtil.isNextEventStartElement(reader, itemWrapperQName));
113     } catch (XMLStreamException ex) {
114       throw new BindingException(ex);
115     }
116 
117     if (groupWrapperQName != null) {
118       consumeEndElement(reader, groupWrapperQName);
119     }
120 
121     collector.applyCollection(obj);
122   }
123 
124   protected JavaTypeAdapter<?> getTypeAdapter() throws BindingException {
125     synchronized (this) {
126       if (typeAdapter == null) {
127         Class<?> itemClass = getPropertyBinding().getPropertyInfo().getItemType();
128         typeAdapter = getBindingContext().getJavaTypeAdapter(itemClass);
129       }
130 
131       return typeAdapter;
132     }
133   }
134 
135   @Override
136   public boolean canConsume(QName nextQName) throws BindingException {
137     return (groupWrapperQName != null && groupWrapperQName.equals(nextQName))
138         || (itemWrapperQName != null && itemWrapperQName.equals(nextQName))
139         || getTypeAdapter().canHandleQName(nextQName);
140   }
141 
142   protected QName getGroupWrapperQName() {
143     PropertyInfo propertyInfo = getPropertyBinding().getPropertyInfo();
144 
145     QName retval = null;
146     if (propertyInfo instanceof CollectionPropertyInfo) {
147       CollectionPropertyInfo/secauto/metaschema/binding/model/property/CollectionPropertyInfo.html#CollectionPropertyInfo">CollectionPropertyInfo collectionPropertyInfo = (CollectionPropertyInfo) propertyInfo;
148 
149       if (XmlGroupAsBehavior.GROUPED.equals(collectionPropertyInfo.getXmlGroupAsBehavior())) {
150         retval = collectionPropertyInfo.getGroupXmlQName();
151       }
152     }
153     return retval;
154   }
155 
156   protected QName getItemWrapperQName() {
157     QName retval;
158     if (!isChildWrappedInXml()) {
159       // TODO: check for non singleton value?
160       retval = null;
161     } else {
162       ModelItemPropertyBinding propertyBinding = getPropertyBinding();
163       retval = propertyBinding.getXmlQName();
164     }
165     return retval;
166   }
167 
168   @Override
169   public boolean isChildWrappedInXml() {
170     ModelItemPropertyBinding propertyBinding = getPropertyBinding();
171     return !(propertyBinding instanceof FieldPropertyBinding)
172         || ((FieldPropertyBinding) propertyBinding).isWrappedInXml()
173         || !MarkupMultiline.class.isAssignableFrom((Class<?>) propertyBinding.getPropertyInfo().getItemType());
174   }
