66    @Override
67    public void writeJson(Object obj, PropertyBindingFilter filter, JsonWritingContext writingContext)
68        throws BindingException {
69      JsonGenerator generator = writingContext.getEventWriter();
70      try {
71        generator.writeStartObject();
72  
73        List<FlagPropertyBinding> flags = FlagUtil.filterFlags(obj, getClassBinding().getFlagPropertyBindings(), filter);
74  
75        if (!flags.isEmpty()) {
76          // Output flags
77          FlagUtil.writeFlags(obj, flags, writingContext);
78        }
79  
80        writeBody(obj, writingContext);
81  
82        generator.writeEndObject();
83      } catch (IOException ex) {
84        throw new BindingException(ex);
85      }
86    }
87  
88    protected void writeBody(Object obj, JsonWritingContext writingContext) throws BindingException, IOException {
89      for (ModelItemPropertyBinding propertyBinding : getClassBinding().getModelItemPropertyBindings()) {
90        PropertyInfo propertyInfo = propertyBinding.getPropertyInfo();
91        Object value = propertyInfo.getValue(obj);
92  
93        if (value != null) {
94          handlePropertyBinding(value, propertyBinding, writingContext);
95        }
96      }
97    }

99    protected void handlePropertyBinding(Object value, ModelItemPropertyBinding propertyBinding,
100       JsonWritingContext writingContext) throws BindingException, IOException {
101     PropertyInfo propertyInfo = propertyBinding.getPropertyInfo();
102     BindingContext bindingContext = writingContext.getBindingContext();
103 
104     // first determine what type of item we need to write
105     Class<?> itemClass = propertyBinding.getPropertyInfo().getItemType();
106     ClassBinding<?> itemClassBinding = bindingContext.getClassBinding(itemClass);
107     ItemBinding<?> itemBinding;
108     if (itemClassBinding != null) {
109       if (itemClassBinding instanceof FieldClassBinding) {
110         itemBinding
111             = new FieldItemBinding((FieldClassBinding<?>) itemClassBinding, (FieldPropertyBinding) propertyBinding);
112       } else if (itemClassBinding instanceof AssemblyClassBinding) {
113         itemBinding = new AssemblyItemBinding((AssemblyClassBinding<?>) itemClassBinding,
114             (AssemblyPropertyBinding) propertyBinding);
115       } else {
116         throw new UnsupportedOperationException(String.format("Unsupported class binding '%s' for class '%s'",
117             itemClassBinding.getClass().getName(), itemClassBinding.getClazz().getName()));
118       }
119     } else {
120       itemBinding = new DataTypeItemBinding(propertyBinding);
121     }
122 
123     if (propertyInfo instanceof CollectionPropertyInfo) {
124       CollectionPropertyInfo/secauto/metaschema/binding/model/property/CollectionPropertyInfo.html#CollectionPropertyInfo">CollectionPropertyInfo collectionPropertyInfo = (CollectionPropertyInfo) propertyInfo;
125       if (collectionPropertyInfo.isList()) { // In this case we need to consider if collapse is used
126         boolean isAllowSingleton
127             = JsonGroupAsBehavior.SINGLETON_OR_LIST.equals(collectionPropertyInfo.getJsonGroupAsBehavior());
128         @SuppressWarnings("unchecked")
129         List<? extends Object> values = (List<? extends Object>) value;
130         if (!values.isEmpty()) {
131           if (itemClassBinding != null) {
132             if (itemClassBinding instanceof FieldClassBinding
133                 && ((FieldClassBinding<?>) itemClassBinding).isCollapsible()) {
134 
135               if (itemClassBinding.getFlagPropertyBindings().isEmpty() || values.size() == 1) {
136                 // no need to collapse a single entry
137                 handleCollection(values, isAllowSingleton, itemBinding, writingContext);
138               } else {
139                 @SuppressWarnings("unchecked")
140                 BoundClassItemBinding<FieldClassBinding<?>, FieldPropertyBinding> fieldItemBinding
141                     = (BoundClassItemBinding<FieldClassBinding<?>, FieldPropertyBinding>) itemBinding;
142                 handleCollapsibleFieldCollection(values, isAllowSingleton, fieldItemBinding, writingContext);
143               }
144             } else {
145               handleCollection(values, isAllowSingleton, itemBinding, writingContext);
146             }
147           } else {
148             handleCollection(values, isAllowSingleton, itemBinding, writingContext);
149           }
150         }
151       } else {
152         // must be a map, which implies use of JsonGroupAsBehavior.KEYED with a bound
153         // class that has a JsonKey
154         @SuppressWarnings("unchecked")
155         Map<String, Object> values = (Map<String, Object>) value;
156         Collection<Object> entries = values.values();
157         if (!values.isEmpty()) {
158           @SuppressWarnings("unchecked")
159           BoundClassItemBinding<ClassBinding<?>, ModelItemPropertyBinding> boundClassItemBinding
160               = (BoundClassItemBinding<ClassBinding<?>, ModelItemPropertyBinding>) itemBinding;
161           handleMapPropertyBinding(entries, boundClassItemBinding, writingContext);
162         }
163       }
164     } else {
165       handleCollection(Collections.singletonList(value), true, itemBinding, writingContext);
166     }
167   }

169   private void handleCollection(List<? extends Object> items, boolean isAllowSingleton, ItemBinding<?> itemBinding,
170       JsonWritingContext writingContext) throws BindingException, IOException {
171 
172     // figure out the field name
173     ModelItemPropertyBinding propertyBinding = itemBinding.getPropertyBinding();
174     String fieldName = propertyBinding.getJsonFieldName(writingContext.getBindingContext());
175 
176     JsonGenerator generator = writingContext.getEventWriter();
177     generator.writeFieldName(fieldName);
178 
179     if (isAllowSingleton && items.size() == 1) {
180       itemBinding.writeValue(items.get(0), null, writingContext);
181     } else {
182       generator.writeStartArray();
183 
184       for (Object item : items) {
185         itemBinding.writeValue(item, null, writingContext);
186       }
187 
188       generator.writeEndArray();
189     }
190   }

192   private void handleMapPropertyBinding(Collection<Object> items,
193       BoundClassItemBinding<ClassBinding<?>, ModelItemPropertyBinding> itemBinding, JsonWritingContext writingContext)
194       throws IOException, BindingException {
195 
196     FlagPropertyBinding jsonKey = itemBinding.getClassBinding().getJsonKeyFlagPropertyBinding();
197     PropertyInfo jsonKeyPropertyInfo = jsonKey.getPropertyInfo();
198 
199     // figure out the field name
200     ModelItemPropertyBinding propertyBinding = itemBinding.getPropertyBinding();
201 
202     String fieldName = propertyBinding.getJsonFieldName(writingContext.getBindingContext());
203 
204     JsonGenerator generator = writingContext.getEventWriter();
205     generator.writeFieldName(fieldName);
206 
207     generator.writeStartObject();
208 
209     for (Object item : items) {
210       String key = jsonKeyPropertyInfo.getValue(item).toString();
211 
212       generator.writeFieldName(key);
213       itemBinding.writeValue(item, (PropertyBinding binding) -> binding.equals(jsonKey), writingContext);
214     }
215 
216     generator.writeEndObject();
217   }

221       throws BindingException, IOException {
222

235       if (itemValues == null) {
236         itemValues = new LinkedList<>();
237         keyToValuesMap.put(key, itemValues);
238       }
239       itemValues.add(value);
240     }
241 
242     // figure out the field name
243     // String fieldName = collectionPropertyInfo.getGroupXmlQName().getLocalPart();
244     String fieldName = itemBinding.getPropertyBinding().getJsonFieldName(writingContext.getBindingContext());
245 
246     JsonGenerator generator = writingContext.getEventWriter();
247     generator.writeFieldName(fieldName);
248 
249     if (isAllowSingleton && keyToValuesMap.size() == 1) {
250       Map.Entry<CollapseKeyBuilder.Key, List<Object>> entry = keyToValuesMap.entrySet().iterator().next();
251       entry.getKey().write(entry.getValue(), writingContext);
252     } else {
253       generator.writeStartArray();
254 
255       for (Map.Entry<CollapseKeyBuilder.Key, List<Object>> entry : keyToValuesMap.entrySet()) {
256         entry.getKey().write(entry.getValue(), writingContext);
257       }
258 
259       generator.writeEndArray();
260     }
