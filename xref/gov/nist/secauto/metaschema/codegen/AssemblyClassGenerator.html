47  public class AssemblyClassGenerator extends AbstractClassGenerator<AssemblyDefinition> {
48    private static final Logger logger = LogManager.getLogger(AssemblyClassGenerator.class);

50    public AssemblyClassGenerator(AssemblyDefinition definition, JavaTypeSupplier supplier) {
51      super(definition, supplier);
52  
53      processModel(getDefinition());
54    }
55  
56    private void processModel(ModelContainer model) {
57      for (InfoElementInstance instance : model.getInstances()) {
58        if (instance instanceof ChoiceInstance) {
59          processModel((ChoiceInstance) instance);
60          continue;
61        }
62  
63        // else the instance is a model instance with a named instance
64        ModelInstance/gov/nist/secauto/metaschema/model/info/instances/ModelInstance.html#ModelInstance">ModelInstance modelInstance = (ModelInstance) instance;
65        newModelInstance(modelInstance);
66      }
67    }
68  
69    public PropertyGenerator newModelInstance(ModelInstance instance) {
70      JavaType itemJavaType;
71      if (instance instanceof FieldInstance) {
72        FieldInstance/gov/nist/secauto/metaschema/model/info/instances/FieldInstance.html#FieldInstance">FieldInstance fieldInstance = (FieldInstance) instance;
73        if (fieldInstance.getDefinition().getFlagInstances().isEmpty()) {
74          DataType dataType = DataType.lookupByDatatype(fieldInstance.getDatatype());
75          // this is a simple value
76          itemJavaType = dataType.getJavaType();
77        } else {
78          itemJavaType = getJavaTypeSupplier().getClassJavaType(instance.getDefinition());
79        }
80      } else if (instance instanceof AssemblyInstance) {
81        itemJavaType = getJavaTypeSupplier().getClassJavaType(instance.getDefinition());
82      } else {
83        String msg = String.format("Unknown model instance type: %s", instance.getClass().getCanonicalName());
84        logger.error(msg);
85        throw new RuntimeException(msg);
86      }
87  
88      PropertyGenerator context;
89      int maxOccurance = instance.getMaxOccurs();
90      if (maxOccurance == -1 || maxOccurance > 1) {
91        if (JsonGroupAsBehavior.KEYED.equals(instance.getJsonGroupAsBehavior())) {
92          context = new MapInstanceGenerator(instance, itemJavaType, this);
93        } else {
94          context = new ListInstanceGenerator(instance, itemJavaType, this);
95        }
96      } else {
97        context = new SingletonInstanceGenerator(instance, itemJavaType, this);
98      }
99  
100     addPropertyGenerator(context);
101     return context;
102   }

104   @Override
105   protected boolean isRootClass() {
106     return Objects.equals(getDefinition(), getDefinition().getContainingMetaschema().getRootAssemblyDefinition());
107   }

109   @Override
110   protected void buildClass(ClassBuilder builder) {
111     super.buildClass(builder);
112 
113     AssemblyDefinition definition = getDefinition();
114     if (Objects.equals(definition, definition.getContainingMetaschema().getRootAssemblyDefinition())) {
115 
116       StringBuilder instanceBuilder = new StringBuilder();
117       instanceBuilder.append("name = \"");
118       instanceBuilder.append(getDefinition().getName());
119       instanceBuilder.append("\"");
120 
121       String namespace = getXmlNamespace().toString();
122       String containingNamespace = getDefinition().getContainingMetaschema().getXmlNamespace().toString();
123       if (!containingNamespace.equals(namespace)) {
124         instanceBuilder.append(", namespace = \"");
125         instanceBuilder.append(namespace);
126         instanceBuilder.append("\"");
127       }
128 
129       builder.annotation(RootWrapper.class, instanceBuilder.toString());
130     }
131   }
